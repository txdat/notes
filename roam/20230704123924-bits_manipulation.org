#+title: bits-manipulation

* represent integer as bit array
#+begin_src C++
array<int, 32> getBitArr(int n) {
    array<int, 32> arr{0};
    if (n == INT_MIN) {
        arr[31] = 1;
        return arr;
    }
    bool isNeg = n < 0;
    n = abs(n);
    int i = 0;
    while (n) {
        arr[i++] = n & 1;
        n >>= 1;
    }
    // if n is negative, bit represetation of n is ~(-n)
    if (isNeg) {
        for (i = 0; i < 32; i++) arr[i] = 1 - arr[i];
    }
    return arr;
}
