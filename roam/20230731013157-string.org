int longestPrefixSuffix(string &s) {
  int n = s.length(), len = 0, i = 1; // len is longest prefix/suffix of s[0..i)
  vector<int> lps(n); // len of longest prefix/suffix of s[0..i]
  lps[0] = 0;
  while (i < n) {
    if (s[len] == s[i]) {
      lps[i++] = ++len;
    } else if (len) {
      len = lps[len - 1];
    } else {
      lps[i++] = 0;
    }
  }
  return len;
}
